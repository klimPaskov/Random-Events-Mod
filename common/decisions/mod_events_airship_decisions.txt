mod_events_airship_decision_category = {
	airship_timeout = {
		available = {
			OR = { 
				hidden_trigger = { is_ai = yes }
			}
		}
		activation = {
			hidden_trigger = { is_ai = no }
			has_global_flag = airship
		}

			hidden_effect = {

						clr_country_flag = airship_country

						limit = { has_country_flag = airship_country NOT = { has_country_flag = airship_above } }
						country_event = { id = mod_events.timed_events.248 }
					}
					every_country = {
						limit = { NOT = { has_country_flag = airship_country } has_country_flag = airship_above }
						country_event = { id = mod_events.timed_events.249 }
					}
					if = { limit = { check_variable = { global.airship_d = 120 } } # 119 is the end destination
						clr_global_flag = airship
						every_country = { limit = { NOT = { tag = USA } } country_event = { id = mod_events.timed_events.188 } }
						USA = { country_event = { id = mod_events.timed_events.235 } }
						news_event = { id = mod_events.news.69 days = 2 }
					}
				}
			}
		}
	}
